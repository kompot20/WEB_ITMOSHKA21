                from numpy import linalg as LA

                def linan(l1, l2):
                  a1 = list(map(int, l1.split()))
                  a2 = list(map(int, l2.split()))
                  b = [a1[2], a2[2]]
                  a = [a1[:-1], a2[:-1]]
                  return LA.solve(a, b)
                ot = linan('1 1 1', '1 2 3')
                print(ot[0], ot[1])
